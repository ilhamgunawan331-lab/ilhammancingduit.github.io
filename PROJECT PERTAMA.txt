<script>
let adminData = [];
let qrisData = [];
let resultData = [];

function readCSV(file, callback){
  let reader = new FileReader();
  reader.onload = e => {
    let rows = e.target.result.split("\n").map(r => r.trim());
    callback(rows);
  };
  reader.readAsText(file);
}

function processData(){
  let f1 = document.getElementById("fileAdmin").files[0];
  let f2 = document.getElementById("fileQris").files[0];

  if(!f1 || !f2) return alert("Upload kedua file!");

  readCSV(f1, data1 => {
    adminData = data1;

    readCSV(f2, data2 => {
      qrisData = data2;

      matchData();
    });
  });
}

function matchData(){
  resultData = [];

  let klop=0, selisih=0, hilang=0;

  adminData.forEach(a=>{
    if(qrisData.includes(a)){
      resultData.push({id:a, status:"KLOP"});
      klop++;
    } else {
      resultData.push({id:a, status:"HILANG"});
      hilang++;
    }
  });

  qrisData.forEach(q=>{
    if(!adminData.includes(q)){
      resultData.push({id:q, status:"SELISIH"});
      selisih++;
    }
  });

  document.getElementById("klop").innerText = klop;
  document.getElementById("selisih").innerText = selisih;
  document.getElementById("hilang").innerText = hilang;
  document.getElementById("total").innerText = resultData.length;

  renderTable(resultData);
  renderChart(klop, selisih, hilang);
}

function renderTable(data){
  let html = "";
  data.forEach(d=>{
    html += `${d.id} - ${d.status}<br>`;
  });
  document.getElementById("result").innerHTML = html;
}

function applyFilter(){
  let key = document.getElementById("search").value.toLowerCase();

  let filtered = resultData.filter(d => d.id.toLowerCase().includes(key));

  renderTable(filtered);
}

function renderChart(klop, selisih, hilang){
  new Chart(document.getElementById("chart"), {
    type: 'doughnut',
    data: {
      labels: ["Klop","Selisih","Hilang"],
      datasets: [{
        data: [klop, selisih, hilang]
      }]
    }
  });
}

function exportData(){
  let csv = "ID,STATUS\n";
  resultData.forEach(d=>{
    csv += `${d.id},${d.status}\n`;
  });

  let blob = new Blob([csv], {type:"text/csv"});
  let a = document.createElement("a");
  a.href = URL.createObjectURL(blob);
  a.download = "hasil.csv";
  a.click();
}
